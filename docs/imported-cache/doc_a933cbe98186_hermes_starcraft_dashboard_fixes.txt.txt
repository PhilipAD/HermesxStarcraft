2. One-Command Auto-Start Script
Agent-Runnable

Create ~/.hermes/starcraft-dashboard/start-empire.sh:

#!/bin/bash
set -e

echo "🚀 Hermes StarCraft Empire Auto-Launch (Agent Mode)"
cd ~/.hermes/starcraft-dashboard

# 1. Start Bridge (port 9121) if not running
if ! lsof -Pi :9121 -sTCP:LISTEN -t >/dev/null 2>&1; then
  echo "→ Starting Bridge (Hermes state sync)..."
  (cd server && npm run dev) &
  BRIDGE_PID=$!
fi

# 2. Start Viewer (port 9120) if not running
if ! lsof -Pi :9120 -sTCP:LISTEN -t >/dev/null 2>&1; then
  echo "→ Starting 3D Viewer (Vite + Three.js)..."
  npm run dev &
  VIEWER_PID=$!
fi

sleep 4

echo "✅ Empire is LIVE!"
echo "🌐 Open → http://127.0.0.1:9120"
echo "📡 Bridge connected to Hermes SQLite (73+ entities mapped)"

# Auto-open browser (works on Linux/Mac/Windows via WSL)
if command -v xdg-open >/dev/null; then
  xdg-open http://127.0.0.1:9120
elif command -v open >/dev/null; then
  open http://127.0.0.1:9120
elif command -v start >/dev/null; then
  start http://127.0.0.1:9120
fi

# Keep script alive (or let agent manage PIDs)
wait

Make it executable:

chmod +x ~/.hermes/starcraft-dashboard/start-empire.sh

Add to package.json scripts in root:

"scripts": {
  "start:empire": "./start-empire.sh",
  "start:prod": "npm run build && serve -s dist -p 9120 --single"
}


3. Docker / Production Polish
Zero-Downtime Restarts

Update docker-compose.yml
Add healthchecks + depends_on:

services:
  bridge:
    build: ./server
    ports: ["9121:9121"]
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9121/health"]
      interval: 5s
      retries: 3

  viewer:
    build: .
    ports: ["9120:9120"]
    depends_on:
      bridge:
        condition: service_healthy
    environment:
      VITE_BRIDGE_URL: "ws://bridge:9121"

Run once:

docker-compose up -d --build
